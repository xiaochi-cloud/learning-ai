## 📈 自动化脚本

### 每小时自动提交

```bash
#!/bin/bash
# hourly-commit.sh

cd /home/admin/.openclaw/workspace/learning-ai

# 检查变更
git status --porcelain

# 如果有变更，自动提交
if [ $? -eq 0 ]; then
    git add -A
    git commit -m "auto: 每小时自动提交 $(date '+%Y-%m-%d %H:%M')"
    git push
fi
```

### 重要信息检测

```bash
#!/bin/bash
# check-important.sh

# 检查关键词
KEYWORDS=("突破" "解决" "机会" "风险" "重要")

# 扫描新增文件
for file in $(git status --porcelain | grep "^??" | cut -c4-); do
    if grep -qE "${KEYWORDS[@]}" "$file" 2>/dev/null; then
        echo "发现重要信息：$file"
        # 发送通知
    fi
done
```
